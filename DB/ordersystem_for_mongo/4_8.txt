db.orders_w_lines.aggregate([
	{
		$group:
		{
			_id: "$customer_id",
			total_order: {$sum: 1}
		}
	},
	{
		$sort: {total_order: -1}
	},
	{
		$limit: 3
	},
	{
		$lookup:
		{
			from: "customer",
			localField: "_id",
			foreignField: "customer_id",
			as: "customer"
		}
	},
	{
		$unwind: "$customer"
	},
	{
		$project:
		{
			_id: 0,
			customer_name: "$customer.customer_name",
			order_count: "$total_order"
		}
	}
])
